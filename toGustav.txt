#include <stdio.h>
#include <time.h>

int main() {
    time_t start_time = time(NULL);
    int x = 7, result;
    
    while (difftime(time(NULL), start_time) < 10) {
        result = x*x*x % 10;
        x++;
    }
    
    printf("Result: %d\n", result);
    return 0;
}
